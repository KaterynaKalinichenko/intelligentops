# Install steps

1. Copy script on your server (CentOS)
2. Make - chmod 777 Script.sh
3. Fill in the fields with parameters
	default_access_key=""  #Your aws access key
	default_secret_key=""  #Your aws secret key
	default_region=""      #Specify region

	image_id=""    		#AMI image id
	instance_type=""   	#Instance type
	instance_count=""	#How many instance your need
	vpc_id=""		#VPC id you want to use
